1 Introduction & Project Overview

A high-level overview of the frontend layer and its design goals.

The frontend of Nexora AI was developed as a standalone, production-ready client layer for an AI-powered Learning Management System (LMS). Built with React 19 and the latest web technologies, the interface serves three distinct user roles — Learner, Instructor, and Admin — each with a dedicated dashboard and tailored experience.

🤖 A. AI Tutor — Pause Detection Logic (Hook)

const useAiPauseDetection = (isPaused, currentTime) => {
  const pauseTimerRef = useRef(null);

  useEffect(() => {
    if (isPaused) {
      // Start AI prompt timer after 3 seconds of inactivity
      pauseTimerRef.current = setTimeout(() => {
        const now = Date.now();
        const approxTime = Math.floor(currentTime / 5) * 5;

        if (now - lastPromptTimeRef.current > 10000 && !promptedTimestamps.current.has(approxTime)) {
          setMessages(prev => [...prev, {
            id: `prompt-${Date.now()}`,
            role: 'ai',
            text: `I noticed you paused. Want me to explain this concept?`,
            isPrompt: true,
            timestamp: currentTime,
          }]);
          lastPromptTimeRef.current = now;
          promptedTimestamps.current.add(approxTime);
        }
      }, 3000);
    } else {
      clearTimeout(pauseTimerRef.current);
    }
    return () => clearTimeout(pauseTimerRef.current);
  }, [isPaused, currentTime]);
};

🌗 D. Dark/Light Theme Toggle

const getInitialTheme = () => {
    const stored = localStorage.getItem('ai_edu_theme');
    if (stored === 'dark' || stored === 'light') return stored;
    return 'dark';
};

export const ThemeProvider = ({ children }) => {
    const [theme, setThemeState] = useState(getInitialTheme);

    useEffect(() => {
        const root = document.documentElement;
        if (theme === 'dark') {
            root.classList.add('dark');
        } else {
            root.classList.remove('dark');
        }
        localStorage.setItem('ai_edu_theme', theme);
    }, [theme]);

    const toggleTheme = () =>
        setThemeState(prev => prev === 'dark' ? 'light' : 'dark');

    return (
        <ThemeContext.Provider value={{ theme, toggleTheme }}>
            {children}
        </ThemeContext.Provider>
    );
};
